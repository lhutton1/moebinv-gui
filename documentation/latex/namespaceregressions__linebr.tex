>>> assert isinstance( numeric(0) != numeric(1), relational) == True

>>> assert isinstance( numeric(0) >= numeric(0), relational) == True

>>> assert isinstance( numeric(0) < numeric(1), relational) == True

>>> assert isinstance( numeric(0) > numeric(-1), relational) == True
